```
FPS=3
DATE=$(date +%Y%m%d)
RUN=${DATE}_${FPS}fps_full
OUT_ROOT="$RUNS_DIR/aixsuture/frames_full_${FPS}fps"
LOG_DIR="$RUNS_DIR/aixsuture/logs"
mkdir -p "$OUT_ROOT" "$LOG_DIR"

{
  for i in $(seq -w 1 11); do
    echo "=== Package_${i} $(date +%H:%M:%S) ==="
    uv run python3 preprocessing.py \
      --data_root "$DATASETS_DIR/Package_${i}" \
      --out_dir "$OUT_ROOT" \
      --run "$RUN" \
      --frame_rate "$FPS" \
    || { echo "FAILED at Package_${i}"; exit 1; }
  done
  echo "preprocess done $(date +%H:%M:%S)"
} 2>&1 | tee "$LOG_DIR/preprocess_${RUN}.log"

FRAMES_DIR="$OUT_ROOT/preprocessing/$RUN/frames_${FPS}fps"
cp "$DATASETS_DIR/OSATS.xlsx" "$FRAMES_DIR/OSATS.xlsx"
find "$FRAMES_DIR" -mindepth 1 -maxdepth 1 -type d | wc -l
```
